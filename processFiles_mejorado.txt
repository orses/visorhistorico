class ProcesadorDeImagenes {
    /**
     * @param {Object} configuracion Objeto de configuración con las dependencias globales necesarias.
     */
    constructor(configuracion) {
        this.contenedorGaleria = configuracion.contenedorGaleria;
        this.gestorMetadatos = configuracion.gestorMetadatos;
        this.diccionarioImagenes = configuracion.diccionarioImagenes;
        this.alCompletar = configuracion.alCompletar;
        this.loteTamano = 10;
        
        this.elementosUI = this._crearInterfazProgreso();
    }

    _crearInterfazProgreso() {
        // En lugar de innerHTML, se instancian los nodos directamente por seguridad y rendimiento
        this.contenedorGaleria.innerHTML = '';
        
        const contenedor = document.createElement('div');
        contenedor.className = 'progress-container';
        
        const titulo = document.createElement('h3');
        titulo.className = 'progress-title';
        titulo.textContent = 'Procesando imágenes...';
        
        const pista = document.createElement('div');
        pista.className = 'progress-track';
        
        const barra = document.createElement('div');
        barra.id = 'progressBar';
        barra.className = 'progress-fill';
        
        const texto = document.createElement('div');
        texto.id = 'progressText';
        texto.className = 'progress-text';
        texto.textContent = 'Iniciando...';
        
        pista.appendChild(barra);
        contenedor.appendChild(titulo);
        contenedor.appendChild(pista);
        contenedor.appendChild(texto);
        
        this.contenedorGaleria.appendChild(contenedor);
        
        return { barra, texto };
    }

    _actualizarProgreso(procesados, total) {
        if (!this.elementosUI.barra || !this.elementosUI.texto) return;
        const porcentaje = Math.round((procesados / total) * 100);
        this.elementosUI.barra.style.width = `${porcentaje}%`;
        this.elementosUI.barra.textContent = `${porcentaje}%`;
        this.elementosUI.texto.textContent = `${procesados} de ${total} imágenes válidas`;
    }

    _pausaVisual() {
        // requestAnimationFrame es más eficiente que setTimeout para sincronizar el reflow visual del navegador
        return new Promise(resolve => requestAnimationFrame(resolve));
    }

    async procesar(archivos) {
        // 1. Validación preventiva de archivos de imagen mediante su tipo Mime
        const archivosValidos = Array.from(archivos).filter(archivo => archivo.type.startsWith('image/'));
        const total = archivosValidos.length;
        
        if (total === 0) {
            console.warn('No se han detectado archivos de imagen válidos.');
            return;
        }

        let procesados = 0;
        
        for (const archivo of archivosValidos) {
            const nombreArchivo = archivo.name;
            
            try {
                const metadatos = this.gestorMetadatos.parseFilename(nombreArchivo);
                
                if (metadatos?.location) {
                    const coordenadas = this.gestorMetadatos.getCoordinates(metadatos.location);
                    if (coordenadas) metadatos.coordinates = coordenadas;
                }
                
                if (metadatos) {
                    this.gestorMetadatos.updateMetadata(nombreArchivo, metadatos);
                }
                
                if (this.diccionarioImagenes[nombreArchivo]) {
                    URL.revokeObjectURL(this.diccionarioImagenes[nombreArchivo]);
                }
                
                this.diccionarioImagenes[nombreArchivo] = URL.createObjectURL(archivo);
                
            } catch (error) {
                console.error(`Error procesando el archivo «${nombreArchivo}»:`, error);
            } finally {
                procesados++;
                this._actualizarProgreso(procesados, total);
                
                if (procesados % this.loteTamano === 0) {
                    await this._pausaVisual();
                }
            }
        }
        
        // Finalización
        if (this.elementosUI.texto) {
            this.elementosUI.texto.textContent = '✓ Procesamiento completado';
            this.elementosUI.texto.classList.add('progress-success');
        }
        
        await new Promise(resolve => setTimeout(resolve, 600));
        
        // 2. Inyección de dependencias para el estado global a través de un callback
        if (typeof this.alCompletar === 'function') {
            const clavesImagenes = Object.keys(this.gestorMetadatos.getAllMetadata());
            this.alCompletar(clavesImagenes);
        }
    }
}
